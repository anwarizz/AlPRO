import java.util.Scanner;

public class GanjilGenap {
    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);

        // Inputkan sebuah bilangan bulat
        System.out.print("Masukkan sebuah bilangan: ");
        int bilangan = input.nextInt();

        // Periksa sisa hasil bagi dengan 2
        if (bilangan % 2 == 0) {
            System.out.println("Bilangan " + bilangan + " adalah bilangan genap.");
        } else {
            System.out.println("Bilangan " + bilangan + " adalah bilangan ganjil.");
        }

        input.close();
    }
}

import java.util.Scanner;

public class KelulusanMahasiswa {
    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);

        // Input nama dan nilai mahasiswa
        System.out.print("Masukkan nama mahasiswa: ");
        String nama = input.nextLine();

        System.out.print("Masukkan nilai mahasiswa: ");
        double nilai = input.nextDouble();

        // Menentukan keterangan lulus atau tidak lulus
        String keterangan;
        if (nilai > 70) {
            keterangan = "Lulus";
        } else {
            keterangan = "Tidak Lulus";
        }

        // Menampilkan hasil
        System.out.println("\n=== Hasil Kelulusan ===");
        System.out.println("Nama       : " + nama);
        System.out.println("Nilai      : " + nilai);
        System.out.println("Keterangan : " + keterangan);

        input.close();
    }
}
